import pandas as pd

# Load Titanic dataset
titanic_data = pd.read_csv('titanic.csv')

# Create a new feature 'Family_Size' by summing 'SibSp' and 'Parch'
titanic_data['Family_Size'] = titanic_data['SibSp'] + titanic_data['Parch']

# Create a new feature 'Traveling_Alone' indicating whether a passenger is traveling alone or with family
titanic_data['Traveling_Alone'] = (titanic_data['Family_Size'] == 0).astype(int)

# Display the first few rows of the dataset with the new features
